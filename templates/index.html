import hashlib

def sha256(text):
    # Produces a fixed 256-bit (64 hex char) digest
    return hashlib.sha256(text.encode()).hexdigest()

# Registration: store only the hash
users_db[username] = sha256(password)

# Login: hash what user typed, compare to stored hash

AES-128 CBC — Practical 5
Symmetric encryption for messages
from Crypto.Cipher import AES
from Crypto.Util.Padding import pad, unpad
import os, base64, hashlib

def aes_encrypt(plaintext, key_str):
    key    = hashlib.sha256(key_str.encode()).digest()[:16] # 16-byte key
    iv     = os.urandom(16)                                  # random IV
    cipher = AES.new(key, AES.MODE_CBC, iv)
    ct     = cipher.encrypt(pad(plaintext.encode(), 16))
    return base64.b64encode(ct), base64.b64encode(iv)

def aes_decrypt(ct_b64, iv_b64, key_str):
    key    = hashlib.sha256(key_str.encode()).digest()[:16]
    cipher = AES.new(key, AES.MODE_CBC, base64.b64decode(iv_b64))
    return unpad(cipher.decrypt(base64.b64decode(ct_b64)), 16).decode()
